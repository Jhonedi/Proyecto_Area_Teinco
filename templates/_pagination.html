    {# ← Anterior #}
    • {# Primera página #}
  • 1
    • {# Puntos suspensivos izquierdos #} {% if w_start > 2 %}
  • …
    • {% endif %} {# Ventana central #} {% for p in range(w_start, w_end + 1) %}
  • {{ p }}
    • {% endfor %} {# Puntos suspensivos derechos #} {% if w_end < total_pages - 1 %}
  • …
    • {% endif %} {# Última página (solo si total_pages > 1, ya garantizado) #}
  • {{ total_pages }}
    • {# → Siguiente #}
{% endif %} {% endmacro %}
